‚öôÔ∏è JavaScript API

Programmatic Control

// Initialize with options const toggle = new ToggleSwitch('#my-toggle', { onChange: (checked, instance) => console.log('Changed:', checked), onEnable: (instance) => console.log('Enabled'), onDisable: (instance) => console.log('Disabled') }); // API Methods toggle.setChecked(true); // Set state toggle.isChecked(); // Get state toggle.toggle(); // Toggle state toggle.enable(); // Enable switch toggle.disable(); // Disable switch toggle.destroy(); // Clean up // Auto-initialize all toggles ToggleSwitch.initAll();
